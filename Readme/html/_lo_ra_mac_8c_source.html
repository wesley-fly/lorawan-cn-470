1720 {
1721  uint8_t i = 0;
1722  uint8_t cmdCount = 0;
1723 
1724  if( ( cmdBufIn == NULL ) || ( cmdBufOut == NULL ) )
1725  {
1726  return 0;
1727  }
1728 
1729  for( i = 0; i < length; i++ )
1730  {
1731  switch( cmdBufIn[i] )
1732  {
1733  // STICKY
1734  case MOTE_MAC_DL_CHANNEL_ANS:
1735  case MOTE_MAC_RX_PARAM_SETUP_ANS:
1736  { // 1 byte payload
1737  cmdBufOut[cmdCount++] = cmdBufIn[i++];
1738  cmdBufOut[cmdCount++] = cmdBufIn[i];
1739  break;
1740  }
1741  case MOTE_MAC_RX_TIMING_SETUP_ANS:
1742  { // 0 byte payload
1743  cmdBufOut[cmdCount++] = cmdBufIn[i];
1744  break;
1745  }
1746  // NON-STICKY
1747  case MOTE_MAC_DEV_STATUS_ANS:
1748  { // 2 bytes payload
1749  i += 2;
1750  break;
1751  }
1752  case MOTE_MAC_LINK_ADR_ANS:

1754  { // 1 byte payload
1755  i++;

1757  }
1758  case MOTE_MAC_TX_PARAM_SETUP_ANS:
1759  case MOTE_MAC_DUTY_CYCLE_ANS:
1760  case MOTE_MAC_LINK_CHECK_REQ:
1761  { // 0 byte payload
1762  break;
1763  }
1764  default:
1765  break;
1766  }
1767  }

1772 static void ProcessMacCommands( uint8_t *payload, uint8_t macIndex, uint8_t commandsSize, uint8_t snr )
1773 {
1774  uint8_t status = 0;
1775 
1776  while( macIndex < commandsSize )
1777  {
1778  // Decode Frame MAC commands
1779  switch( payload[macIndex++] )
1780  {
1781  case SRV_MAC_LINK_CHECK_ANS:
1782  MlmeConfirm.Status = LORAMAC_EVENT_INFO_STATUS_OK;
1783  MlmeConfirm.DemodMargin = payload[macIndex++];
1784  MlmeConfirm.NbGateways = payload[macIndex++];
1785  break;
1786  case SRV_MAC_LINK_ADR_REQ:
1787  {
1788  LinkAdrReqParams_t linkAdrReq;
1789  int8_t linkAdrDatarate = DR_0;
1790  int8_t linkAdrTxPower = TX_POWER_0;
1791  uint8_t linkAdrNbRep = 0;
1792  uint8_t linkAdrNbBytesParsed = 0;
1793 
1794  // Fill parameter structure
1795  linkAdrReq.Payload = &payload[macIndex - 1];
1796  linkAdrReq.PayloadSize = commandsSize - ( macIndex - 1 );
1797  linkAdrReq.AdrEnabled = AdrCtrlOn;
1798  linkAdrReq.UplinkDwellTime = LoRaMacParams.UplinkDwellTime;
1799  linkAdrReq.CurrentDatarate = LoRaMacParams.ChannelsDatarate;
1800  linkAdrReq.CurrentTxPower = LoRaMacParams.ChannelsTxPower;
1801  linkAdrReq.CurrentNbRep = LoRaMacParams.ChannelsNbRep;
1802 
1803  // Process the ADR requests
1804  status = RegionLinkAdrReq( LoRaMacRegion, &linkAdrReq, &linkAdrDatarate,
1805  &linkAdrTxPower, &linkAdrNbRep, &linkAdrNbBytesParsed );
1806 
1807  if( ( status & 0x07 ) == 0x07 )
1808  {
1809  LoRaMacParams.ChannelsDatarate = linkAdrDatarate;
1810  LoRaMacParams.ChannelsTxPower = linkAdrTxPower;
1811  LoRaMacParams.ChannelsNbRep = linkAdrNbRep;
1812  }
1813 
1814  // Add the answers to the buffer
1815  for( uint8_t i = 0; i < ( linkAdrNbBytesParsed / 5 ); i++ )
1816  {
1817  AddMacCommand( MOTE_MAC_LINK_ADR_ANS, status, 0 );
1818  }
1819  // Update MAC index
1820  macIndex += linkAdrNbBytesParsed - 1;
1821  }
1822  break;
1823  case SRV_MAC_DUTY_CYCLE_REQ:
1824  MaxDCycle = payload[macIndex++];
1825  AggregatedDCycle = 1 << MaxDCycle;
1826  AddMacCommand( MOTE_MAC_DUTY_CYCLE_ANS, 0, 0 );
1827  break;
1828  case SRV_MAC_RX_PARAM_SETUP_REQ:
1829  {
1830  RxParamSetupReqParams_t rxParamSetupReq;
1831  status = 0x07;
1832 
1833  rxParamSetupReq.DrOffset = ( payload[macIndex] >> 4 ) & 0x07;
1834  rxParamSetupReq.Datarate = payload[macIndex] & 0x0F;
1835  macIndex++;
1836 
1837  rxParamSetupReq.Frequency = ( uint32_t )payload[macIndex++];
1838  rxParamSetupReq.Frequency |= ( uint32_t )payload[macIndex++] << 8;
1839  rxParamSetupReq.Frequency |= ( uint32_t )payload[macIndex++] << 16;
1840  rxParamSetupReq.Frequency *= 100;
1841 
1842  // Perform request on region
1843  status = RegionRxParamSetupReq( LoRaMacRegion, &rxParamSetupReq );
1844 
1845  if( ( status & 0x07 ) == 0x07 )
1846  {
1847  LoRaMacParams.Rx2Channel.Datarate = rxParamSetupReq.Datarate;
1848  LoRaMacParams.Rx2Channel.Frequency = rxParamSetupReq.Frequency;
1849  LoRaMacParams.Rx1DrOffset = rxParamSetupReq.DrOffset;
1850  }
1851  AddMacCommand( MOTE_MAC_RX_PARAM_SETUP_ANS, status, 0 );
1852  }
1853  break;
1854  case SRV_MAC_DEV_STATUS_REQ:
1855  {
1856  uint8_t batteryLevel = BAT_LEVEL_NO_MEASURE;
1857  if( ( LoRaMacCallbacks != NULL ) && ( LoRaMacCallbacks->GetBatteryLevel != NULL ) )
1858  {
1859  batteryLevel = LoRaMacCallbacks->GetBatteryLevel( );
1860  }
1861  AddMacCommand( MOTE_MAC_DEV_STATUS_ANS, batteryLevel, snr & 0x3F);
1862  break;
1863  }
1864  case SRV_MAC_NEW_CHANNEL_REQ:
1865  {
1866  NewChannelReqParams_t newChannelReq;
1867  ChannelParams_t chParam;
1868  status = 0x03;
1869 
1870  newChannelReq.ChannelId = payload[macIndex++];
1871  newChannelReq.NewChannel = &chParam;
1872 
1873  chParam.Frequency = ( uint32_t )payload[macIndex++];
1874  chParam.Frequency |= ( uint32_t )payload[macIndex++] << 8;
1875  chParam.Frequency |= ( uint32_t )payload[macIndex++] << 16;
1876  chParam.Frequency *= 100;
1877  chParam.Rx1Frequency = 0;
1878  chParam.DrRange.Value = payload[macIndex++];
1879 
1880  status = RegionNewChannelReq( LoRaMacRegion, &newChannelReq );
1881 
1882  AddMacCommand( MOTE_MAC_NEW_CHANNEL_ANS, status, 0 );
1883  }
1884  break;
1885  case SRV_MAC_RX_TIMING_SETUP_REQ:
1886  {
1887  uint8_t delay = payload[macIndex++] & 0x0F;
1888 
1889  if( delay == 0 )
1890  {
1891  delay++;
1892  }
1893  LoRaMacParams.ReceiveDelay1 = delay * 1000;
1894  LoRaMacParams.ReceiveDelay2 = LoRaMacParams.ReceiveDelay1 + 1000;
1895  AddMacCommand( MOTE_MAC_RX_TIMING_SETUP_ANS, 0, 0 );
1896  }
1897  break;
1898  case SRV_MAC_TX_PARAM_SETUP_REQ:
1899  {
1900  TxParamSetupReqParams_t txParamSetupReq;
1901  uint8_t eirpDwellTime = payload[macIndex++];
1902 
1903  txParamSetupReq.UplinkDwellTime = 0;
1904  txParamSetupReq.DownlinkDwellTime = 0;
1905 
1906  if( ( eirpDwellTime & 0x20 ) == 0x20 )
1907  {
1908  txParamSetupReq.DownlinkDwellTime = 1;
1909  }
1910  if( ( eirpDwellTime & 0x10 ) == 0x10 )
1911  {
1912  txParamSetupReq.UplinkDwellTime = 1;
1913  }
1914  txParamSetupReq.MaxEirp = eirpDwellTime & 0x0F;
1915 
1916  // Check the status for correctness
1917  if( RegionTxParamSetupReq( LoRaMacRegion, &txParamSetupReq ) != -1 )
1918  {
1919  // Accept command
1920  LoRaMacParams.UplinkDwellTime = txParamSetupReq.UplinkDwellTime;
1921  LoRaMacParams.DownlinkDwellTime = txParamSetupReq.DownlinkDwellTime;
1922  LoRaMacParams.MaxEirp = LoRaMacMaxEirpTable[txParamSetupReq.MaxEirp];
1923  // Add command response
1924  AddMacCommand( MOTE_MAC_TX_PARAM_SETUP_ANS, 0, 0 );
1925  }
1926  }
1927  break;
1928  case SRV_MAC_DL_CHANNEL_REQ:
1929  {
1930  DlChannelReqParams_t dlChannelReq;
1931  status = 0x03;
1932 
1933  dlChannelReq.ChannelId = payload[macIndex++];
1934  dlChannelReq.Rx1Frequency = ( uint32_t )payload[macIndex++];
1935  dlChannelReq.Rx1Frequency |= ( uint32_t )payload[macIndex++] << 8;
1936  dlChannelReq.Rx1Frequency |= ( uint32_t )payload[macIndex++] << 16;
1937  dlChannelReq.Rx1Frequency *= 100;
1938 
1939  status = RegionDlChannelReq( LoRaMacRegion, &dlChannelReq );
1940 
1941  AddMacCommand( MOTE_MAC_DL_CHANNEL_ANS, status, 0 );
1942  }
1943  break;
1944  default:
1945  // Unknown command. ABORT MAC commands processing
1946  return;
1947  }
1948  }
1949 }
