Description: This example demonstrate how to join to LoRaWAN.

/*
(C)2017 Gemtek
Description: This example demonstrate how to join to LoRaWAN.
OTAA Flow
0. Read OTAA configurations of DevEUI, AppEUI, AppKey from EEPROM
1. Initila OTAA parameters to LoRaWAN by using function "LoRaMacMlmeRequest" in DEVICE_STATE_JOIN status
2. Add timer "JoinTimer" for periodically check if join is successful or not,
if not go back to DEVICE_STATE_JOIN status and join again.
3. If join flow is successful, change status to DEVICE_STATE_JOIN_SUCCESS then show the device address from gateway.
ABP Flow
0. Initial ABP configurations by using Gemtek API of GemtekLoRaMacKeyInit
1. Initila ABP parameters to LoRaWAN
Note:
Write OTAA or ABP configurations can refer example of WriteLoRaWANConfig
*/
#include <stdio.h>
#include <string.h>
#include "board.h"
#include "board-config.h"
#include "delay.h"
#include "LoRaMac.h"
#include "Comissioning.h"
#include "LoRaMacCrypto.h"
#include "LoRaMacTest.h"
#ifdef REGION_AS923
#define ACTIVE_REGION LORAMAC_REGION_AS923
#elif REGION_AU915
#define ACTIVE_REGION LORAMAC_REGION_AU915
#elif REGION_CN470
#define ACTIVE_REGION LORAMAC_REGION_CN470
#elif REGION_CN779
#define ACTIVE_REGION LORAMAC_REGION_CN779
#elif REGION_EU433
#define ACTIVE_REGION LORAMAC_REGION_EU433
#elif REGION_EU868
#define ACTIVE_REGION LORAMAC_REGION_EU868
#elif REGION_KR920
#define ACTIVE_REGION LORAMAC_REGION_KR920
#elif REGION_IN865
#define ACTIVE_REGION LORAMAC_REGION_IN865
#elif REGION_US915
#define ACTIVE_REGION LORAMAC_REGION_US915
#elif REGION_US915_HYBRID
#define ACTIVE_REGION LORAMAC_REGION_US915_HYBRID
#else
#warning "No active region defined."
#endif
#ifndef ACTIVE_REGION
#warning "No active region defined, LORAMAC_REGION_EU868 will be used as default."
#define ACTIVE_REGION LORAMAC_REGION_EU868
#endif
#define FIFO_TX_SIZE 128
#define FIFO_RX_SIZE 128
char str[128];
uint8_t UartTxBuffer[FIFO_TX_SIZE];
uint8_t UartRxBuffer[FIFO_RX_SIZE];
static void McpsConfirm( McpsConfirm_t *mcpsConfirm ){};
static void McpsIndication( McpsIndication_t *mcpsIndication ){};
static void MlmeConfirm( MlmeConfirm_t *mlmeConfirm ){};
static void MlmeIndication( MlmeIndication_t *MlmeIndication ){};
static TimerEvent_t TxNextPacketTimer;
void OnTxNextPacketTimerEvent( void );
bool SendFrame(void);
static enum eDeviceState
{
DEVICE_STATE_JOIN,
DEVICE_STATE_JOIN_SUCCESS,
DEVICE_STATE_SLEEP,
DEVICE_STATE_SENDFRAME,
}DeviceState;
#if( OVER_THE_AIR_ACTIVATION != 0 )
static TimerEvent_t JoinTimer;
static uint8_t DevEui[8];
static uint8_t AppEui[8];
static uint8_t AppKey[16];
static void OnJoinTimerEvent( void )
{
MibRequestConfirm_t mibReq;
LoRaMacStatus_t status;
TimerStop( &JoinTimer );
mibReq.Type = MIB_NETWORK_JOINED;
status = LoRaMacMibGetRequestConfirm( &mibReq );
if( status == LORAMAC_STATUS_OK )
{
//Check if join success or not.
if( mibReq.Param.IsNetworkJoined == true )
{
DeviceState = DEVICE_STATE_JOIN_SUCCESS;
}
else
{
DeviceState = DEVICE_STATE_JOIN;
}
}
}
#endif
int main( void )
{
LoRaMacPrimitives_t LoRaMacPrimitives;
LoRaMacCallback_t LoRaMacCallbacks;
BoardInitMcu( );
/* Initializes the UART object*/
FifoInit( &Uart1.FifoTx, UartTxBuffer, FIFO_TX_SIZE );
FifoInit( &Uart1.FifoRx, UartRxBuffer, FIFO_RX_SIZE );
UartInit( &Uart1, UART_1, UART_TX, UART_RX );
UartConfig( &Uart1, RX_TX, 9600, UART_8_BIT, UART_1_STOP_BIT, NO_PARITY, NO_FLOW_CTRL );
/*Initial LoRaWan Callback function*/
LoRaMacPrimitives.MacMcpsConfirm = McpsConfirm;
LoRaMacPrimitives.MacMcpsIndication = McpsIndication;
LoRaMacPrimitives.MacMlmeConfirm = MlmeConfirm;
LoRaMacPrimitives.MacMlmeIndication = MlmeIndication;
LoRaMacCallbacks.GetBatteryLevel = BoardGetBatteryLevel;
LoRaMacInitialization( &LoRaMacPrimitives, &LoRaMacCallbacks, ACTIVE_REGION);
#ifdef REGION_EU868
LoRaMacTestSetDutyCycleOn(false);
#endif
TimerInit( &TxNextPacketTimer, OnTxNextPacketTimerEvent );
#if( OVER_THE_AIR_ACTIVATION != 0 )
//Start OTAA Join Flow
//Initial timer for join flow
TimerInit( &JoinTimer, OnJoinTimerEvent );
// Get the OTAA config DevEui, AppEui and AppKey from EEPROM
if(GemtekGetOTAAConfig(DevEui,AppEui,AppKey) == 0)
{
//Get successful
sprintf(str, "Get OTAA configurations successful!\r\n");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
}
else
{
//Get failed
sprintf(str, "Get OTAA configurations failed!\r\n");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
while(1);
}
//Start Join flow
DeviceState = DEVICE_STATE_JOIN;
while( 1 )
{
switch( DeviceState )
{
case DEVICE_STATE_JOIN:
{
MlmeReq_t mlmeReq;
sprintf(str, "Start Joining FLow\r\n");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
mlmeReq.Type = MLME_JOIN;
mlmeReq.Req.Join.DevEui = DevEui;
mlmeReq.Req.Join.AppEui = AppEui;
mlmeReq.Req.Join.AppKey = AppKey;
// mlmeReq.Req.Join.NbTrials = 3;
LoRaMacMlmeRequest( &mlmeReq );
/*Add timer for checking join is success or not every 10 seconds*/
TimerSetValue( &JoinTimer, 10000 );
TimerStart( &JoinTimer );
DeviceState = DEVICE_STATE_SLEEP;
break;
}
case DEVICE_STATE_JOIN_SUCCESS:
{
MibRequestConfirm_t mibGet;
mibGet.Type = MIB_DEV_ADDR;
LoRaMacMibGetRequestConfirm(&mibGet);
sprintf(str, "Join Successful! Device Addr:%0x08\r\n",mibGet.Param.DevAddr);
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
DeviceState = DEVICE_STATE_SENDFRAME;
break;
}
case DEVICE_STATE_SENDFRAME:
{
SendFrame();
DeviceState = DEVICE_STATE_SLEEP;
TimerSetValue( &TxNextPacketTimer, 10000 );
TimerStart( &TxNextPacketTimer );
break;
}
case DEVICE_STATE_SLEEP:
{
DelayMs(1);
break;
}
default:
DeviceState = DEVICE_STATE_SLEEP;
break;
}
}
#else
//Start ABP Join Flow
uint32_t DevAddr;
MibRequestConfirm_t mibReq;
//Initial ABP configurations from EEPROM
if(GemtekLoRaMacKeyInit( ) == 0)
{
sprintf(str, "Init ABP configurations successful!\r\n");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
}
else
{
sprintf(str, "Init ABP configurations failed!\r\n");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
while(1);
}
//Get device address from EEPROM
DevAddr = GemtekGetDevAddr( );
mibReq.Type = MIB_NET_ID;
mibReq.Param.NetID = LORAWAN_NETWORK_ID;
LoRaMacMibSetRequestConfirm( &mibReq );
//Update device address to LoRaWAN
mibReq.Type = MIB_DEV_ADDR;
mibReq.Param.DevAddr = DevAddr;
LoRaMacMibSetRequestConfirm( &mibReq );
mibReq.Type = MIB_NETWORK_JOINED;
mibReq.Param.IsNetworkJoined = true;
LoRaMacMibSetRequestConfirm( &mibReq );
//Join success!
sprintf(str, "Join flow completed, device address:%X!\r\n",DevAddr);
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
DeviceState = DEVICE_STATE_SENDFRAME;
while(1)
{
switch(DeviceState)
{
case DEVICE_STATE_SLEEP:
{
DelayMs(1);
break;
}
case DEVICE_STATE_SENDFRAME:
{
SendFrame();
DeviceState = DEVICE_STATE_SLEEP;
TimerSetValue( &TxNextPacketTimer, 10000 );
TimerStart( &TxNextPacketTimer );
break;
}
default:
DeviceState = DEVICE_STATE_SLEEP;
break;
}
}
#endif
}
bool SendFrame()
{
McpsReq_t mcpsReq;
uint8_t appData[2];
memset(appData,0,2);
appData[0] = 0x94;
appData[1] = 0x53;
mcpsReq.Type = MCPS_UNCONFIRMED;
mcpsReq.Req.Unconfirmed.fPort = 1;
mcpsReq.Req.Unconfirmed.fBuffer = appData;
mcpsReq.Req.Unconfirmed.fBufferSize = 2;
mcpsReq.Req.Unconfirmed.Datarate = 0;
if( LoRaMacMcpsRequest( &mcpsReq ) == LORAMAC_STATUS_OK )
{
sprintf(str, "Send Frame\r\n");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
return false;
}
return true;
}
void OnTxNextPacketTimerEvent( void )
{
TimerStop( &TxNextPacketTimer );
DeviceState = DEVICE_STATE_SENDFRAME;
}
