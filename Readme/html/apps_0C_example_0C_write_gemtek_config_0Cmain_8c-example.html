This example describes how to configure OTAA or ABP parameters by using Gemtek API.

/*

Description: This example describes how to configure OTAA or ABP parameters by using Gemtek API.
*/
#include <stdio.h>
#include <string.h>
#include "board.h"
#include "board-config.h"
#include "LoRaMac.h"
#include "LoRaMacCrypto.h"
#define FIFO_TX_SIZE 512
#define FIFO_RX_SIZE 512
uint8_t UartTxBuffer[FIFO_TX_SIZE];
uint8_t UartRxBuffer[FIFO_RX_SIZE];
void ConfigureABP(void);
void ConfigureOTAA(void);
int main( void )
{
BoardInitMcu( );
/* Initial UART1*/
FifoInit( &Uart1.FifoTx, UartTxBuffer, FIFO_TX_SIZE );
FifoInit( &Uart1.FifoRx, UartRxBuffer, FIFO_RX_SIZE );
UartInit( &Uart1, UART_1, UART_TX, UART_RX );
UartConfig( &Uart1, RX_TX, 9600, UART_8_BIT, UART_1_STOP_BIT, NO_PARITY, NO_FLOW_CTRL );
/*
* Write ABP configurations Example
*/
ConfigureABP();
/*
* Write OTAA Configurations Example
*/
ConfigureOTAA();
while(1);
}
void ConfigureABP()
{
char str[128];
uint32_t devAddr = 0x10000003;
uint8_t nwkSKeyHex[16] = {0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0x55, 0x66};
uint8_t appSKeyHex[16] = {0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0x94, 0x53};
uint32_t devAddrTmp;
uint16_t nwkKeyCrc, appKeyCrc;
/*Get current ABP confiurations from EEPROM*/
sprintf(str, "Get current ABP configurations!\r\n");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
if(GemtekGetABPConfig(&devAddrTmp, &nwkKeyCrc, &appKeyCrc) == 0)
{
sprintf(str, "devAddr:%X, nwkSkeyCrc:%X, appSKeyCrc:%X\r\n",devAddr, nwkKeyCrc, appKeyCrc);
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
}
else
{
sprintf(str, "Get current ABP configurations failed!!!\r\n");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
}
/*Write ABP configurations to EEPROM*/
sprintf(str, "Write ABP configurations to EEPROM\r\n");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
if(GemtekWriteABPConfigHex(devAddr, nwkSKeyHex, appSKeyHex) == 0)
{
sprintf(str, "Write successful!\r\n");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
if(GemtekGetABPConfig(&devAddrTmp, &nwkKeyCrc, &appKeyCrc) == 0)
{
sprintf(str, "The new ABP confiurations\r\n");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
sprintf(str, "devAddr:%X, nwkSkeyCrc:%X, appSKeyCrc:%X\r\n",devAddrTmp, nwkKeyCrc, appKeyCrc);
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
sprintf(str, "Check the wriet values are correct or not\r\n");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
/*Check write value is correct or not*/
if(devAddr == devAddrTmp)
{
sprintf(str, "devAddr OK!\r\n");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
}
if(CrcStd16((const uint8_t*)nwkSKeyHex,16) == nwkKeyCrc)
{
sprintf(str, "nwkKeyCrc OK!\r\n");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
}
if(CrcStd16((const uint8_t*)appSKeyHex,16) == appKeyCrc)
{
sprintf(str, "appKeyCrc OK!\r\n");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
}
}
}
else
{
sprintf(str, "Write failed\r\n");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
}
sprintf(str, "\r\n\r\n");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
}
void ConfigureOTAA()
{
char str[128];
int i;
uint8_t devEuiTmp[8];
uint8_t appEuiTmp[8];
uint8_t appKeyTmp[16];
uint8_t devEuiHex[8] = {0x20, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x20};
uint8_t appEuiHex[8] = {0x10, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x20};
uint8_t appKeyHex[16] = {0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0xaa, 0x11, 0x22};
/*Get current OTAA configurations*/
sprintf(str, "Get current OTAA configurations!\r\n");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
if(GemtekGetOTAAConfig(devEuiTmp,appEuiTmp,appKeyTmp)==0)
{
sprintf(str, "DevEui:");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
for(i=0;i<8;i++)
{
sprintf(str, "%X ",devEuiTmp[i]);
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
}
sprintf(str, "\r\nAppEui:");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
for(i=0;i<8;i++)
{
sprintf(str, "%X ",appEuiTmp[i]);
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
}
sprintf(str, "\r\nAppKey:");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
for(i=0;i<16;i++)
{
sprintf(str, "%X ",appKeyTmp[i]);
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
}
sprintf(str, "\r\n");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
}
else
{
sprintf(str, "Get OTAA configurations failed!\r\n");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
}
/*Write OTAA configurations to EEPROM*/
sprintf(str, "Write OTAA configurations to EEPROM\r\n");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
if(GemtekWriteOTAAConfigHex(devEuiHex,appEuiHex,appKeyHex)==0)
{
sprintf(str, "Write successful!!!\r\n");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
/*Get current OTAA configurations*/
sprintf(str, "Get new OTAA configurations!\r\n");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
if(GemtekGetOTAAConfig(devEuiTmp,appEuiTmp,appKeyTmp)==0)
{
sprintf(str, "DevEui:");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
for(i=0;i<8;i++)
{
sprintf(str, "%X ",devEuiTmp[i]);
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
}
sprintf(str, "\r\nAppEui:");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
for(i=0;i<8;i++)
{
sprintf(str, "%X ",appEuiTmp[i]);
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
}
sprintf(str, "\r\nAppKey:");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
for(i=0;i<16;i++)
{
sprintf(str, "%X ",appKeyTmp[i]);
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
}
sprintf(str, "\r\n");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
}
else
{
sprintf(str, "Get failed!\r\n");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
}
}
else
{
sprintf(str, "Write failed!\r\n");
UartPutBuffer(&Uart1, (uint8_t *)str, strlen(str));
}
