1259 /*
1260  * Rx window precise timing
1261  *
1262  * For more details please consult the following document, chapter 3.1.2.
1263  * https://www.semtech.com/uploads/documents/SX1272_settings_for_LoRaWAN_v2.0.pdf
1264  * or
1265  * https://www.semtech.com/uploads/documents/SX1276_settings_for_LoRaWAN_v2.0.pdf
1266  *
1267  * Downlink start: T = Tx + 1s (+/- 20 us)
1268  * |
1269  * TRxEarly | TRxLate
1270  * | | |
1271  * | | +---+---+---+---+---+---+---+---+
1272  * | | | Latest Rx window |
1273  * | | +---+---+---+---+---+---+---+---+
1274  * | | |
1275  * +---+---+---+---+---+---+---+---+
1276  * | Earliest Rx window |
1277  * +---+---+---+---+---+---+---+---+
1278  * |
1279  * +---+---+---+---+---+---+---+---+
1280  *Downlink preamble 8 symbols | | | | | | | | |
1281  * +---+---+---+---+---+---+---+---+
1282  *
1283  * Worst case Rx window timings
1284  *
1285  * TRxLate = DEFAULT_MIN_RX_SYMBOLS * tSymbol - RADIO_WAKEUP_TIME
1286  * TRxEarly = 8 - DEFAULT_MIN_RX_SYMBOLS * tSymbol - RxWindowTimeout - RADIO_WAKEUP_TIME
1287  *
1288  * TRxLate - TRxEarly = 2 * DEFAULT_SYSTEM_MAX_RX_ERROR
1289  *
1290  * RxOffset = ( TRxLate + TRxEarly ) / 2
1291  *
1292  * RxWindowTimeout = ( 2 * DEFAULT_MIN_RX_SYMBOLS - 8 ) * tSymbol + 2 * DEFAULT_SYSTEM_MAX_RX_ERROR
1293  * RxOffset = 4 * tSymbol - RxWindowTimeout / 2 - RADIO_WAKE_UP_TIME
1294  *
1295  * Minimal value of RxWindowTimeout must be 5 symbols which implies that the system always tolerates at least an error of 1.5 * tSymbol
1296  */
